16  /**
17   * ed25519 implementation of {@link AbstractPrivateKey}
18   *
19   * @author Alexandre Carvalho
20   * @author Andre Bertolace
21   * @since 0.1.0
22   */
23  @NoArgsConstructor
24  @EqualsAndHashCode(callSuper = true)
25  public class Ed25519PrivateKey extends AbstractPrivateKey {
26  
27      private Ed25519PrivateKeyParameters privateKeyParameters;
28  
29      public Ed25519PrivateKey(byte[] privateKey) {
30          super(privateKey);
31          loadPrivateKey(privateKey);
32      }
33  
34      @Override
35      public void loadPrivateKey(byte[] privateKey) {
36          privateKeyParameters = new Ed25519PrivateKeyParameters(privateKey, 0);
37      }
38  
39      /*
40       * SEQUENCE (3 elem) INTEGER 0 SEQUENCE (1 elem) OBJECT IDENTIFIER 1.3.101.112
41       * curveEd25519 (EdDSA 25519 signature algorithm) OCTET STRING (32 byte)
42       * 38AECE974291F14B5FEF97E1B21F684394120B6E7A8AFB04398BBE787E8BC559 OCTET STRING
43       * (32 byte) 38AECE974291F14B5FEF97E1B21F684394120B6E7A8AFB04398BBE787E8BC559
44       */
45      @Override
46      public void readPrivateKey(String filename) throws IOException {
47          ASN1Primitive key = ASN1Primitive.fromByteArray(PemFileHelper.readPemFile(filename));
48          PrivateKeyInfo keyInfo = PrivateKeyInfo.getInstance(key);
49          String algoId = keyInfo.getPrivateKeyAlgorithm().getAlgorithm().toString();
50          if (algoId.equals(ASN1Identifiers.Ed25519OID.getId())) {
51              privateKeyParameters = new Ed25519PrivateKeyParameters(keyInfo.getPrivateKey().getEncoded(), 4);
52              setKey(privateKeyParameters.getEncoded());
53          }
54      }
55  
56      @Override
57      public void writePrivateKey(String filename) throws IOException {
58          DERSequence derPrefix = new DERSequence(ASN1Identifiers.Ed25519OID);
59          DEROctetString key = new DEROctetString(new DEROctetString(getKey()));
60          ASN1EncodableVector vector = new ASN1EncodableVector();
61          vector.add(new ASN1Integer(0));
62          vector.add(derPrefix);
63          vector.add(key);
64          DERSequence derKey = new DERSequence(vector);
65          PemFileHelper.writePemFile(filename, derKey.getEncoded(), ASN1Identifiers.PRIVATE_KEY_DER_HEADER);
66      }
67  
68      @Override
69      public byte[] sign(byte[] message) {
70          Signer signer = new Ed25519Signer();
71          signer.init(true, privateKeyParameters);
72          signer.update(message, 0, message.length);
73          byte[] signature;
74          try {
75              signature = signer.generateSignature();
76              return signature;
77          } catch (DataLengthException | CryptoException e) {
78              // TODO: throw new SomeException();
79              return null;
80          }
81      }
82  
83      @Override
84      public AbstractPublicKey derivePublicKey() {
85          return new Ed25519PublicKey(privateKeyParameters.generatePublicKey().getEncoded());
86      }
87  }
