3   import org.bouncycastle.crypto.digests.SHA256Digest;
4   
5   /**
6    * Sha256 Hash helper class
7    *
8    * @author Alexandre Carvalho
9    * @author Andre Bertolace
10   * @since 0.2.0
11   */
12  public final class Sha256 {
13  
14      /**
15       * returns a Sha 256 Hash of size length in bytes
16       *
17       * @param input byte array to hash
18       * @return a byte array of size 'length' bytes
19       */
20      public static byte[] digest(byte[] input) {
21          SHA256Digest d = new SHA256Digest();
22          d.update(input, 0, input.length);
23          byte[] result = new byte[d.getDigestSize()];
24          d.doFinal(result, 0);
25          return result;
26      }
27  }
