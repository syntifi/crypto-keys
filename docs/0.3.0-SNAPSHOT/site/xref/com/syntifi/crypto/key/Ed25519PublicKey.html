13  /**
14   * ed25519 implementation of {@link AbstractPublicKey}
15   *
16   * @author Alexandre Carvalho
17   * @author Andre Bertolace
18   * @since 0.1.0
19   */
20  @NoArgsConstructor
21  @EqualsAndHashCode(callSuper = true)
22  public class Ed25519PublicKey extends AbstractPublicKey {
23  
24      private Ed25519PublicKeyParameters publicKeyParameters;
25  
26      public Ed25519PublicKey(byte[] publicKey) {
27          super(publicKey);
28          loadPublicKey(publicKey);
29      }
30  
31      @Override
32      public void loadPublicKey(byte[] publicKey) {
33          publicKeyParameters = new Ed25519PublicKeyParameters(publicKey, 0);
34      }
35  
36      /*
37       * SEQUENCE (2 elem) SEQUENCE (1 elem) OBJECT IDENTIFIER 1.3.101.112
38       * curveEd25519 (EdDSA 25519 signature algorithm) BIT STRING (256 bit) <KEY
39       * HERE>
40       */
41      @Override
42      public void readPublicKey(String filename) throws IOException {
43          ASN1Primitive derKey = ASN1Primitive.fromByteArray(PemFileHelper.readPemFile(filename));
44          ASN1Sequence objBaseSeq = ASN1Sequence.getInstance(derKey);
45          String objId = ASN1ObjectIdentifier
46                  .getInstance(ASN1Sequence.getInstance(objBaseSeq.getObjectAt(0)).getObjectAt(0)).getId();
47          if (objId.equals(ASN1Identifiers.Ed25519OID.getId())) {
48              DERBitString key = DERBitString.getInstance(objBaseSeq.getObjectAt(1));
49              publicKeyParameters = new Ed25519PublicKeyParameters(key.getBytes(), 0);
50              setKey(publicKeyParameters.getEncoded());
51          }
52      }
53  
54      @Override
55      public void writePublicKey(String filename) throws IOException {
56          DERSequence derPrefix = new DERSequence(ASN1Identifiers.Ed25519OID);
57          DERBitString key = new DERBitString(getKey());
58          ASN1EncodableVector vector = new ASN1EncodableVector();
59          vector.add(derPrefix);
60          vector.add(key);
61          DERSequence derKey = new DERSequence(vector);
62          PemFileHelper.writePemFile(filename, derKey.getEncoded(), ASN1Identifiers.PUBLIC_KEY_DER_HEADER);
63      }
64  
65      @Override
66      public Boolean verify(byte[] message, byte[] signature) {
67          Signer verifier = new Ed25519Signer();
68          verifier.init(false, publicKeyParameters);
69          verifier.update(message, 0, message.length);
70  
71          return verifier.verifySignature(signature);
72      }
73  }
