4   import org.bouncycastle.crypto.digests.SHA512Digest;
5   import org.bouncycastle.crypto.macs.HMac;
6   import org.bouncycastle.crypto.params.KeyParameter;
7   
8   import java.io.ByteArrayOutputStream;
9   import java.io.IOException;
10  import java.nio.ByteBuffer;
11  import java.util.Arrays;
12  
13  /**
14   * The procedure to implement BIP32 or SLIP 10 to generate deterministic keys hierarchically
15   * https://github.com/bitcoin/bips/blob/master/bip-0032.mediawiki
16   * https://github.com/satoshilabs/slips/blob/master/slip-0010.md
17   *
18   * @author Alexandre Carvalho
19   * @author Andre Bertolace
20   * @since 0.3.0
21   */
22  public class HierarchicalDeterministicKey {
23      private final static Long MAX_VALUE_INDEX = 2147483648L;
24  
25      /**
26       *
27       * @param seed           byte array containing the seed
28       * @param init           chaincode, that is the rightmost 32 bytes of the key
29       * @param derivationPath path to follow when deriving the key
30       * @return derived key in byte array
31       * @throws IOException if an error occurs when processing the bytestream
32       */
33      public static byte[] getFromSeed(byte[] seed, byte[] init, int[] derivationPath) throws IOException {
34          byte[] key = HierarchicalDeterministicKey.getMasterKeyFromSeed(seed, init);
35          byte[] iL;
36          byte[] iR;
37          iL = Arrays.copyOfRange(key, 0, 32);
38          iR = Arrays.copyOfRange(key, 32, 64);
39          for (int i : derivationPath) {
40              key = HierarchicalDeterministicKey.childKeyDerivation(
41                      iL, iR, HierarchicalDeterministicKey.longToBytes(MAX_VALUE_INDEX + i));
42              iL = Arrays.copyOfRange(key, 0, 32);
43              iR = Arrays.copyOfRange(key, 32, 64);
44          }
45          return key;
46      }
47  
48      /**
49       * Derives the masterkey given a seed byte array
50       *
51       * @param seed byte array containing the seed
52       * @param key initial Hmac value
53       * @return master key as a byte array
54       */
55      public static byte[] getMasterKeyFromSeed(byte[] seed, byte[] key) {
56          HMac hMac = new HMac(new SHA512Digest());
57          hMac.init(new KeyParameter(key));
58          hMac.update(seed, 0, seed.length);
59          byte[] result = new byte[hMac.getMacSize()];
60          hMac.doFinal(result, 0);
61          return result;
62      }
63  
64      /**
65       * CKD function in Bipt32 and Slip10
66       *
67       * @param key        master key, that is the leftmost 32 bytes of the key
68       * @param chainCode  hierarchical chain code 'a/b/c' but in array format: { a, b, c}
69       * @param init       chaincode, that is the rightmost 32 bytes of the key
70       * @return master key as a byte array
71       * @throws IOException if an error occurs when processing the bytestream
72       */
73      public static byte[] childKeyDerivation(byte[] key, byte[] chainCode, byte[] init) throws IOException {
74          ByteArrayOutputStream os = new ByteArrayOutputStream();
75          os.write(Hex.decode("00"));
76          os.write(key);
77          os.write(Arrays.copyOfRange(init, 4, 8));
78          HMac hMac = new HMac(new SHA512Digest());
79          hMac.init(new KeyParameter(chainCode));
80          hMac.update(os.toByteArray(), 0, os.size());
81          byte[] result = new byte[hMac.getMacSize()];
82          hMac.doFinal(result, 0);
83          return result;
84      }
85  
86      private static byte[] longToBytes(long x) {
87          ByteBuffer buffer = ByteBuffer.allocate(Long.BYTES);
88          buffer.putLong(x);
89          return buffer.array();
90      }
91  }
