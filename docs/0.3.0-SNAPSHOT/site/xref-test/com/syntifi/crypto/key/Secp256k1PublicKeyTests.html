19  /**
20   * Tests for {@link Secp256k1PublicKey}

26  public class Secp256k1PublicKeyTests extends AbstractCryptoTests {
27      private static final Logger LOGGER = LoggerFactory.getLogger(Secp256k1PublicKeyTests.class);
28  
29      @Test
30      void readPublicKey_should_load_public_key() throws IOException, URISyntaxException {
31          Secp256k1PublicKey pubKey = new Secp256k1PublicKey();
32          String filePath = getResourcesKeyPath("secp256k1/public_key.pem");
33          pubKey.readPublicKey(filePath);
34  
35          assertNotNull(pubKey.getKey());
36      }
37  
38      @Test
39      void writePublicKey_should_equal_source_file() throws URISyntaxException, IOException {
40          Secp256k1PublicKey pubKey = new Secp256k1PublicKey();
41          String filePath = getResourcesKeyPath("secp256k1/public_key.pem");
42          pubKey.readPublicKey(filePath);
43  
44          DateFormat df = new SimpleDateFormat("yyyyMMdd-HHmmss");
45          File publicKeyFile = File.createTempFile(df.format(new Date()), "-public-key-test.pem");
46          pubKey.writePublicKey(publicKeyFile.getPath());
47  
48          assertTrue(compareTextFiles(new File(getResourcesKeyPath("secp256k1/public_key.pem")),
49                  publicKeyFile));
50      }
51  
52      @Test
53      void verify_should_be_ok() throws URISyntaxException, IOException, GeneralSecurityException {
54          String hexSignature = "ea5b38fd0db5fb3d871c47fde1fa4c4db75d1a9e1c0ac54d826e178ee0e63707176b4e63b4f838bd031f007fffd6a4f71d920a10c48ea53dd1573fa2b58a829e";
55  
56          Secp256k1PublicKey pubKey = new Secp256k1PublicKey();
57          String filePath = getResourcesKeyPath("secp256k1/public_key.pem");
58          pubKey.readPublicKey(filePath);
59          LOGGER.info(Hex.toHexString(pubKey.getKey()));
60  
61          assertTrue(pubKey.verify("Test message".getBytes(), Hex.decode(hexSignature)));
62      }
