23  public class MnemonicCodeTest extends AbstractCryptoTests {
24      private static final Logger LOGGER = LoggerFactory.getLogger(MnemonicCodeTest.class);
25  
26      @Test
27      void getSeedFromWordlist_seed_should_match() throws IOException, MnemonicException.MnemonicWordException, MnemonicException.MnemonicChecksumException, MnemonicException.MnemonicLengthException {
28          String words = "army van defense carry jealous true garbage claim echo media make crunch";
29          List<String> wordList = Arrays.asList(words.split(" "));
30          MnemonicCode mnemonicCode = new MnemonicCode("english");
31          byte[] seed = mnemonicCode.toSeed(wordList, "");
32          assertEquals("5b56c417303faa3fcba7e57400e120a0ca83ec5a4fc9ffba757fbe63fbd77a89a1a3be4c67196f57c39a88b76373733891bfaba16ed27a813ceed498804c0570", Hex.encode(seed));
33      }
34  
35      @Test
36      void getEntropyFromWordlist_entropy_should_match() throws IOException, MnemonicException.MnemonicWordException, MnemonicException.MnemonicChecksumException, MnemonicException.MnemonicLengthException {
37          String words = "army van defense carry jealous true garbage claim echo media make crunch";
38          List<String> wordList = Arrays.asList(words.split(" "));
39          MnemonicCode mnemonicCode = new MnemonicCode("english");
40          byte[] entropy = mnemonicCode.toEntropy(wordList);
41          assertEquals("0c1e24e5917779d297e14d45f14e1a1a", Hex.encode(entropy));
42      }
43  
44      @Test
45      void getWordListFromEntropy_words_should_match() throws IOException, MnemonicException.MnemonicWordException, MnemonicException.MnemonicChecksumException, MnemonicException.MnemonicLengthException {
46          MnemonicCode mnemonicCode = new MnemonicCode("english");
47          List<String> w = mnemonicCode.toMnemonic(Hex.decode("0c1e24e5917779d297e14d45f14e1a1a"));
48          String words = "army van defense carry jealous true garbage claim echo media make crunch";
49          List<String> wordList = Arrays.asList(words.split(" "));
50          assertEquals(w, wordList);
51      }
52  
53      @Test
54      void generateRandomList_should_generate_12_words() throws IOException, MnemonicException.MnemonicWordException, MnemonicException.MnemonicChecksumException, MnemonicException.MnemonicLengthException {
55          List<String> words = MnemonicCode.generateSecureRandomWords("english");
56          assertEquals(12, words.size());
57      }
