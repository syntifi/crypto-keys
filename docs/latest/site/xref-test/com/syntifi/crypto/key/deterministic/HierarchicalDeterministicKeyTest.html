19  
20      @Test
21      void getMasterKeyFromSeed_seed_should_match() throws IOException {
22          byte[] init = "ed25519 seed".getBytes(StandardCharsets.UTF_8);
23          assertEquals("a89c4655ab993cf786e4a8517899f0e5ddbdb35ebedc05da341b510bf0c5b8902eb38e1fb071b5a75f0e3b176ddea874ce81fafd6c44cd74c3c381ac5a7ed9fc",
24                  Hex.encode(HierarchicalDeterministicKey.getMasterKeyFromSeed(
25                          "randomkey".getBytes(StandardCharsets.UTF_8), init)));
26      }
27  
28      @Test
29      void getKeyFromSeed_seed_should_match() throws IOException {
30          MnemonicCode mnemonicCode = new MnemonicCode("english");
31          String words = "shoot island position soft burden budget tooth cruel issue economy destroy above";
32          byte[] seed = mnemonicCode.toSeed(Arrays.asList(words.split(" ")), "");
33          assertEquals("577cd910aede2582668a741d476b45e7998e905a4286f701b87b25923501f9d4ea19513b460bcccbc069ebbe4327a59af3d6463045c4b6fa21a5e7004ccfcc3e",
34                  Hex.encode(seed));
35          byte[] init = "ed25519 seed".getBytes(StandardCharsets.UTF_8);
36          assertEquals("17ffa27bec941a557d88ef8d491171eab1dcb147b24b8c7f034766b5a2c425ab" +
37                  "6f1eef3d85f72c2500f3d42cad8632725e830d20ffa61d8dfda1c961c84302f0",
38                  Hex.encode(HierarchicalDeterministicKey.getMasterKeyFromSeed(seed, init)));
39          int[] path = {44, 397, 0};
40          assertEquals("88793a8eeec537c67ee8d459f1899a47a2f1b752d06a4c793c66fd751df80498",
41                  Hex.encode(HierarchicalDeterministicKey.getFromSeed(seed, init, path)));
42          assertTrue(true);
43      }
