21  /**
22   * Tests for {@link Ed25519PrivateKey}

28  public class Ed25519PrivateKeyTests extends AbstractCryptoTests {
29      private static final Logger LOGGER = LoggerFactory.getLogger(Ed25519PrivateKeyTests.class);
30  
31      @Test
32      void readPrivateKey_should_load_private_key() throws IOException, URISyntaxException {
33          Ed25519PrivateKey privateKey = readPrivateKey("ed25519/secret_key.pem");
34          assertNotNull(privateKey.getKey());
35      }
36  
37      @Test
38      void readPrivateKey_derived_public_key_should_equal_generated() throws IOException, URISyntaxException {
39          Ed25519PrivateKey privateKey = readPrivateKey("ed25519/secret_key.pem");
40  
41          // Compare derived public key to generated hex without leading id byte
42          Path hexKeyFilePath = Paths.get(getResourcesKeyPath("ed25519/public_key_hex"));
43          String hexKey = new String(Files.readAllBytes(hexKeyFilePath));
44          LOGGER.debug("Derived public hex Key from {}: {}", hexKeyFilePath,
45                  Hex.encode(privateKey.derivePublicKey().getKey()));
46  
47          assertEquals(hexKey.substring(2), Hex.encode(privateKey.derivePublicKey().getKey()));
48      }
49  
50      @Test
51      void writePrivateKey_should_equal_source_file() throws URISyntaxException, IOException {
52          Ed25519PrivateKey privateKey = readPrivateKey("ed25519/secret_key.pem");
53  
54          DateFormat df = new SimpleDateFormat("yyyyMMdd-HHmmss");
55          File privateKeyFile = File.createTempFile(df.format(new Date()), "-private-key-test.pem");
56  
57          LOGGER.debug("Writing private key to {}", privateKeyFile.getPath());
58          privateKey.writePrivateKey(privateKeyFile.getPath());
59  
60          assertTrue(compareTextFiles(new File(getResourcesKeyPath("ed25519/secret_key.pem")),
61                  privateKeyFile));
62      }
63  
64      @Test
65      void sign_should_sign_message() throws URISyntaxException, IOException {
66          Ed25519PrivateKey privateKey = readPrivateKey("ed25519/secret_key.pem");
67  
68          byte[] signature = privateKey.sign("Test message".getBytes());
69  
70          assertEquals(
71                  "4555103678684364a98478112ce0c298ed841d806d2b67b09e8f0215cc738f3c5a1fca5beaf0474ff636613821bcb97e88b3b4d700e65c6cf7574489e09f170c",
72                  Hex.encode(signature));
73  
74          LOGGER.debug("Signed as {}", signature);
75      }
76  
77      @Test
78      void derived_public_key_should_match_generated() {
79          String base58GeneratedPrivateKey = "32UfEkBGTFpfu6M7RebN1JqMDrdf1YyztgYmcUG5XcRkEraJioFZLPtBvYVmAVvnjWAToSsWScJYSFViv8MaATRF";
80          String base58GeneratedPublicKey = "F8jARHGZdHqnwrxrnv1pFVzzirXZR2vJzeYbvwQbxZyP";
81  
82          Ed25519PrivateKey pk = new Ed25519PrivateKey(Base58.decode(base58GeneratedPrivateKey));
83  
84          assertEquals(base58GeneratedPublicKey, Base58.encode(pk.derivePublicKey().getKey()));
85      }
86  
87      private Ed25519PrivateKey readPrivateKey(String privateKeyPath) throws URISyntaxException, IOException {
88          Ed25519PrivateKey privateKey = new Ed25519PrivateKey();
89          String keyFilePath = getResourcesKeyPath(privateKeyPath);
90          LOGGER.debug("Reading key from {}", keyFilePath);
91          privateKey.readPrivateKey(keyFilePath);
92          LOGGER.debug("Key: {}", Hex.encode(privateKey.getKey()));
93          return privateKey;
94      }
