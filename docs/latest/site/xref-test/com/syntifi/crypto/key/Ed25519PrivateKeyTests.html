36      @Test
37      void readPrivateKey_should_load_private_key() throws IOException, URISyntaxException {
38          Ed25519PrivateKey privateKey = readPrivateKey("ed25519/secret_key.pem");
39          assertNotNull(privateKey.getKey());

42      @Test
43      void readPrivateKey_derived_public_key_should_equal_generated() throws IOException, URISyntaxException {
44          Ed25519PrivateKey privateKey = readPrivateKey("ed25519/secret_key.pem");
45  
46          // Compare derived public key to generated hex without leading id byte
47          Path hexKeyFilePath = Paths.get(getResourcesKeyPath("ed25519/public_key_hex"));
48          String hexKey = new String(Files.readAllBytes(hexKeyFilePath));
49          LOGGER.debug("Derived public hex Key from {}: {}", hexKeyFilePath,
50                  Hex.encode(privateKey.derivePublicKey().getKey()));
51  
52          assertEquals(hexKey.substring(2), Hex.encode(privateKey.derivePublicKey().getKey()));
53      }

55      @Test
56      void writePrivateKey_should_equal_source_file() throws URISyntaxException, IOException {
57          Ed25519PrivateKey privateKey = readPrivateKey("ed25519/secret_key.pem");
58  
59          DateFormat df = new SimpleDateFormat("yyyyMMdd-HHmmss");
60          File privateKeyFile = File.createTempFile(df.format(new Date()), "-private-key-test.pem");
61  
62          LOGGER.debug("Writing private key to {}", privateKeyFile.getPath());
63          privateKey.writePrivateKey(privateKeyFile.getPath());
64  
65          assertTrue(compareTextFiles(new File(getResourcesKeyPath("ed25519/secret_key.pem")),
66                  privateKeyFile));
67      }

82      @Test
83      void derived_public_key_should_match_generated() {
84          String base58GeneratedPrivateKey = "32UfEkBGTFpfu6M7RebN1JqMDrdf1YyztgYmcUG5XcRkEraJioFZLPtBvYVmAVvnjWAToSsWScJYSFViv8MaATRF";
85          String base58GeneratedPublicKey = "F8jARHGZdHqnwrxrnv1pFVzzirXZR2vJzeYbvwQbxZyP";
86  
87          Ed25519PrivateKey pk = new Ed25519PrivateKey(Base58.decode(base58GeneratedPrivateKey));
88  
89          assertEquals(base58GeneratedPublicKey, Base58.encode(pk.derivePublicKey().getKey()));
90      }

92      @Test
93      void create_privateKey_from_seed() throws IOException {
94          MnemonicCode mnemonicCode = new MnemonicCode("english");
95          String words =  "shoot island position soft burden budget tooth cruel issue economy destroy above";
96          byte[] seed = mnemonicCode.toSeed(Arrays.asList(words.split(" ")), "");
97          Ed25519PrivateKey pk1 = Ed25519PrivateKey.deriveFromSeed(seed);
98  
99          byte[] init = "ed25519 seed".getBytes(StandardCharsets.UTF_8);
100         int[] path = {44, 397, 0};
101         byte[] key = HierarchicalDeterministicKey.getFromSeed(seed, init, path);
102         Ed25519PrivateKey pk2 = new Ed25519PrivateKey(key);
103         assertEquals("88793a8eeec537c67ee8d459f1899a47a2f1b752d06a4c793c66fd751df80498",
104                 Hex.encode(pk1.getKey()));
105         assertEquals(Hex.encode(pk2.getKey()), Hex.encode(pk1.getKey()));
106         assertEquals("0c91f6106ff835c0195d5388565a2d69e25038a7e23d26198f85caf6594117ec",
107                 Hex.encode(pk1.derivePublicKey().getKey()));
108         assertEquals(Hex.encode(pk2.derivePublicKey().getKey()), Hex.encode(pk1.derivePublicKey().getKey()));
109     }

111     private Ed25519PrivateKey readPrivateKey(String privateKeyPath) throws URISyntaxException, IOException {
112         Ed25519PrivateKey privateKey = new Ed25519PrivateKey();
113         String keyFilePath = getResourcesKeyPath(privateKeyPath);
114         LOGGER.debug("Reading key from {}", keyFilePath);
115         privateKey.readPrivateKey(keyFilePath);
116         LOGGER.debug("Key: {}", Hex.encode(privateKey.getKey()));
117         return privateKey;
118     }
