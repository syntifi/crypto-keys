15  /**
16   * The procedure to implement BIP32 or SLIP 10 to generate deterministic keys hierarchically
17   * https://github.com/bitcoin/bips/blob/master/bip-0032.mediawiki
18   * https://github.com/satoshilabs/slips/blob/master/slip-0010.md
19   *
20   * @author Alexandre Carvalho
21   * @author Andre Bertolace
22   * @since 0.2.0
23   */
24  public class HierarchicalDeterministicKey {
25      private final static Long MAX_VALUE_INDEX = 2147483648L;
26  
27      public static byte[] getFromSeed(byte[] seed, byte[] init, int[] derivationPath) throws IOException {
28          byte[] result = HierarchicalDeterministicKey.getMasterKeyFromSeed(seed, init);
29          byte[] key;
30          byte[] chainCode;
31          key = Arrays.copyOfRange(result, 0, 32);
32          chainCode = Arrays.copyOfRange(result, 32, 64);
33          for (int i : derivationPath) {
34              byte[] keyi = HierarchicalDeterministicKey.childKeyDerivation(
35                      key, chainCode, HierarchicalDeterministicKey.longToBytes(MAX_VALUE_INDEX + i));
36              key = Arrays.copyOfRange(keyi, 0, 32);
37              chainCode = Arrays.copyOfRange(keyi, 32, 64);
38          }
39          return key;
40      }
41  
42      /**
43       *
44       * @param seed bytes
45       * @param key initial Hmac value
46       * @return byte array
47       */
48      public static byte[] getMasterKeyFromSeed(byte[] seed, byte[] key) {
49          HMac hMac = new HMac(new SHA512Digest());
50          hMac.init(new KeyParameter(key));
51          hMac.update(seed, 0, seed.length);
52          byte[] result = new byte[hMac.getMacSize()];
53          hMac.doFinal(result, 0);
54          return result;
55      }
56  
57      public static byte[] childKeyDerivation(byte[] key, byte[] chainCode, byte[] index) throws IOException {
58          ByteArrayOutputStream os = new ByteArrayOutputStream();
59          os.write(Hex.decode("00"));
60          os.write(key);
61          os.write(Arrays.copyOfRange(index, 4, 8));
62          HMac hMac = new HMac(new SHA512Digest());
63          hMac.init(new KeyParameter(chainCode));
64          hMac.update(os.toByteArray(), 0, os.size());
65          byte[] result = new byte[hMac.getMacSize()];
66          hMac.doFinal(result, 0);
67          return result;
68      }
69  
70      private static byte[] longToBytes(long x) {
71          ByteBuffer buffer = ByteBuffer.allocate(Long.BYTES);
72          buffer.putLong(x);
73          return buffer.array();
74      }
75  }
