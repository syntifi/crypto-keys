50  /**
51   * A MnemonicCode object may be used to convert between binary seed values and
52   * lists of words per <a href="https://github.com/bitcoin/bips/blob/master/bip-0039.mediawiki">the BIP 39
53   * specification</a>
54   * Original implementation at:
55   * https://github.com/bitcoinj/bitcoinj/blob/master/core/src/main/java/org/bitcoinj/crypto/MnemonicCode.java
56   *
57   * @author Alexandre Carvalho
58   * @author Andre Bertolace
59   * @since 0.2.0
60   */
61  
62  public class MnemonicCode {
63      private static final Map<String, String> checkSum = new HashMap<String, String>() {{
64          put("english", "ad90bf3beb7b0eb7e5acd74727dc0da96e0a280a258354e7293fb7e211ac03db");
65      }};
66      private static final Map<String, Charset> charSet = new HashMap<String, Charset>() {{
67          put("english", StandardCharsets.UTF_8);
68      }};
69      private static final int PBKDF2_ROUNDS = 2048;
70      private final List<String> wordList;
71  
72      /**
73       * Creates an MnemonicCode object, initializing with words read from the supplied input stream.
74       * If a wordListDigest is supplied the digest of the words will be checked.
75       *
76       * @param language words languages
77       * @throws IOException
78       * @throws IllegalArgumentException
79       */
80      public MnemonicCode(String language) throws IOException, IllegalArgumentException {
81          InputStream wordStream = getClass().getResourceAsStream("/" + language + ".txt");
82          if (wordStream == null)
83              throw new FileNotFoundException(language);
84          String checksum = checkSum.get(language);
85          Charset charset = charSet.get(language);
86          try (BufferedReader br = new BufferedReader(new InputStreamReader(wordStream, charset))) {
87              this.wordList = br.lines()
88                      .collect(Collectors.collectingAndThen(Collectors.toList(), Collections::unmodifiableList));
89          }
90  
91          if (this.wordList.size() != 2048)
92              throw new IllegalArgumentException("input stream did not contain 2048 bytes");
93  
94          // If a wordListDigest is supplied check to make sure it matches.
95          if (checksum != null) {
96              StringBuilder stringBuilder = new StringBuilder();
97              for (String s : this.getWordList()) {
98                  stringBuilder.append(s);
99              }
100             byte[] digest = Sha256.digest(String.valueOf(stringBuilder).getBytes(charset));
101             String hexDigest = Hex.encode(digest);
102             if (!hexDigest.equals(checksum))
103                 throw new IllegalArgumentException("wordlist checksum mismatch");
104         }
105     }

107     /**
108      * Convert mnemonic word list to seed.
109      *
110      * @param words list of words
111      * @param passphrase password, use "" if not required
112      * @return
113      */
114     public byte[] toSeed(List<String> words, String passphrase) {
115         if (passphrase == null)
116             throw new RuntimeException("A null passphrase is not allowed.");
117 
118         // To create binary seed from mnemonic, we use PBKDF2 function
119         // with mnemonic sentence (in UTF-8) used as a password and
120         // string "mnemonic" + passphrase (again in UTF-8) used as a
121         // salt. Iteration count is set to 2048 and HMAC-SHA512 is
122         // used as a pseudo-random function. Desired length of the
123         // derived key is 512 bits (= 64 bytes).
124         //
125         String pass = String.join(" ", words);
126         String salt = "mnemonic" + passphrase;
127 
128         final PKCS5S2ParametersGenerator pbkdf2 = new PKCS5S2ParametersGenerator(new SHA512Digest());
129         pbkdf2.init(
130                 pass.getBytes(StandardCharsets.UTF_8),
131                 salt.getBytes(StandardCharsets.UTF_8),
132                 PBKDF2_ROUNDS);
133 
134         final KeyParameter key = (KeyParameter) pbkdf2.generateDerivedParameters(512);
135         return key.getKey();
136     }

138     private boolean[] bytesToBits(byte[] data) {
139         boolean[] bits = new boolean[data.length * 8];
140         for (int i = 0; i < data.length; ++i)
141             for (int j = 0; j < 8; ++j)
142                 bits[(i * 8) + j] = (data[i] & (1 << (7 - j))) != 0;
143         return bits;
144     }

156     /**
157      * Convert mnemonic word list to original entropy value.
158      *
159      * @param words list of words
160      * @return entropy byte array
161      * @throws MnemonicException.MnemonicLengthException
162      * @throws MnemonicException.MnemonicWordException
163      * @throws MnemonicException.MnemonicChecksumException
164      */
165     public byte[] toEntropy(List<String> words) throws MnemonicException.MnemonicLengthException,
166             MnemonicException.MnemonicWordException, MnemonicException.MnemonicChecksumException {
167         if (words.size() % 3 > 0)
168             throw new MnemonicException.MnemonicLengthException("Word list size must be multiple of three words.");
169 
170         if (words.size() == 0)
171             throw new MnemonicException.MnemonicLengthException("Word list is empty.");
172 
173         // Look up all the words in the list and construct the
174         // concatenation of the original entropy and the checksum.
175         //
176         int concatLenBits = words.size() * 11;
177         boolean[] concatBits = new boolean[concatLenBits];
178         int wordindex = 0;
179         for (String word : words) {
180             // Find the words index in the wordlist.
181             int ndx = Collections.binarySearch(this.wordList, word);
182             if (ndx < 0)
183                 throw new MnemonicException.MnemonicWordException(word);
184 
185             // Set the next 11 bits to the value of the index.
186             for (int ii = 0; ii < 11; ++ii)
187                 concatBits[(wordindex * 11) + ii] = (ndx & (1 << (10 - ii))) != 0;
188             ++wordindex;
189         }
190 
191         int checksumLengthBits = concatLenBits / 33;
192         int entropyLengthBits = concatLenBits - checksumLengthBits;
193 
194         // Extract original entropy as bytes.
195         byte[] entropy = new byte[entropyLengthBits / 8];
196         for (int ii = 0; ii < entropy.length; ++ii)
197             for (int jj = 0; jj < 8; ++jj)
198                 if (concatBits[(ii * 8) + jj])
199                     entropy[ii] |= 1 << (7 - jj);
200 
201         // Take the digest of the entropy.
202         byte[] hash = Sha256.digest(entropy);
203         boolean[] hashBits = bytesToBits(hash);
204 
205         // Check all the checksum bits.
206         for (int i = 0; i < checksumLengthBits; ++i)
207             if (concatBits[entropyLengthBits + i] != hashBits[i])
208                 throw new MnemonicException.MnemonicChecksumException();
209 
210         return entropy;
211     }

213     /**
214      * Convert entropy data to mnemonic word list.
215      *
216      * @param entropy byte array
217      * @return list of words
218      * @throws MnemonicException.MnemonicLengthException
219      */
220     public List<String> toMnemonic(byte[] entropy) throws MnemonicException.MnemonicLengthException {
221         if (entropy.length % 4 > 0)
222             throw new MnemonicException.MnemonicLengthException("Entropy length not multiple of 32 bits.");
223 
224         if (entropy.length == 0)
225             throw new MnemonicException.MnemonicLengthException("Entropy is empty.");
226 
227         // We take initial entropy of ENT bits and compute its
228         // checksum by taking first ENT / 32 bits of its SHA256 hash.
229 
230         byte[] hash = Sha256.digest(entropy);
231         boolean[] hashBits = bytesToBits(hash);
232 
233         boolean[] entropyBits = bytesToBits(entropy);
234         int checksumLengthBits = entropyBits.length / 32;
235 
236         // We append these bits to the end of the initial entropy.
237         boolean[] concatBits = new boolean[entropyBits.length + checksumLengthBits];
238         System.arraycopy(entropyBits, 0, concatBits, 0, entropyBits.length);
239         System.arraycopy(hashBits, 0, concatBits, entropyBits.length, checksumLengthBits);
240 
241         // Next we take these concatenated bits and split them into
242         // groups of 11 bits. Each group encodes number from 0-2047
243         // which is a position in a wordlist.  We convert numbers into
244         // words and use joined words as mnemonic sentence.
245 
246         ArrayList<String> words = new ArrayList<>();
247         int nWords = concatBits.length / 11;
248         for (int i = 0; i < nWords; ++i) {
249             int index = 0;
250             for (int j = 0; j < 11; ++j) {
251                 index <<= 1;
252                 if (concatBits[(i * 11) + j])
253                     index |= 0x1;
254             }
255             words.add(this.wordList.get(index));
256         }
257 
258         return words;
259     }

261     /**
262      * Check to see if a mnemonic word list is valid.
263      *
264      * @param words list of words
265      * @throws MnemonicException
266      */
267     public void check(List<String> words) throws MnemonicException {
268         toEntropy(words);
269     }

271     /**
272      *
273      * @param language
274      * @return
275      * @throws IOException
276      * @throws MnemonicException.MnemonicLengthException
277      */
278     public static List<String> generateSecureRandomWords(String language) throws IOException, MnemonicException.MnemonicLengthException {
279         MnemonicCode mnemonicCode = new MnemonicCode(language);
280         SecureRandom rnd = new SecureRandom();
281         byte[] entropy = new byte[16];
282         rnd.nextBytes(entropy);
283         return mnemonicCode.toMnemonic(entropy);
284     }
