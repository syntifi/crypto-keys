11  import java.io.FileWriter;
12  import java.io.IOException;
13  import java.math.BigInteger;
14  import java.util.Arrays;
15  
16  /**
17   * secp256k1 implementation of {@link AbstractPrivateKey}
18   *
19   * @author Alexandre Carvalho
20   * @author Andre Bertolace
21   * @since 0.1.0
22   */
23  @Getter
24  public class Secp256k1PrivateKey extends AbstractPrivateKey {
25      private ECKeyPair keyPair;
26  
27      @Override
28      public void readPrivateKey(String filename) throws IOException {
29          ASN1Sequence key = (ASN1Sequence) ASN1Primitive.fromByteArray(PemFileHelper.readPemFile(filename));
30          String algoId = key.getObjectAt(2).toString();
31          if (algoId.equals("[0]" + ASN1Identifiers.Secp256k1OIDCurve) && key.getObjectAt(0).toString().equals("1")) {
32              DEROctetString pk = (DEROctetString) key.getObjectAt(1);
33              keyPair = ECKeyPair.create(pk.getOctets());
34              this.setKey(keyPair.getPrivateKey().toByteArray());
35          }
36      }
37  
38      @Override
39      public void writePrivateKey(String filename) throws IOException {
40          try (FileWriter fileWriter = new FileWriter(filename)) {
41              DERTaggedObject derPrefix = new DERTaggedObject(0, ASN1Identifiers.Secp256k1OIDCurve);
42              DEROctetString key = new DEROctetString(getKey());
43              ASN1EncodableVector vector = new ASN1EncodableVector();
44              vector.add(new ASN1Integer(1));
45              vector.add(key);
46              vector.add(derPrefix);
47              DERSequence derKey = new DERSequence(vector);
48              PemFileHelper.writePemFile(fileWriter, derKey.getEncoded(), ASN1Identifiers.EC_PRIVATE_KEY_DER_HEADER);
49          }
50      }
51  
52      /*
53       * When encoded in DER, this becomes the following sequence of bytes:
54       * <p>
55       * 0x30 b1 0x02 b2 (vr) 0x02 b3 (vs)
56       * <p>
57       * where:
58       * <p>
59       * b1 is a single byte value, equal to the length, in bytes, of the remaining
60       * list of bytes (from the first 0x02 to the end of the encoding);
61       * b2 is a single byte value, equal to the length, in bytes, of (vr);
62       * b3 is a single byte value, equal to the length, in bytes, of (vs);
63       * (vr) is the signed big-endian encoding of the value "r
64       * <p>
65       * ", of minimal length;
66       * (vs) is the signed big-endian encoding of the value "s
67       * ", of minimal length.
68       */
69      @Override
70      public String sign(String message) {
71          SignatureData signature = Sign.signMessage(Hash.sha256(message.getBytes()), keyPair, false);
72          return Hex.toHexString(signature.getR()) + Hex.toHexString(signature.getS());
73      }
74  
75      /*
76       * Returns a Secp256k1PublicKey object in a compressed format
77       * adding the prefix 02/03 to identify the positive or negative Y followed
78       * by the X value in the elliptic curve
79       */
80      @Override
81      public AbstractPublicKey derivePublicKey() {
82          BigInteger pubKey = keyPair.getPublicKey();
83          String pubKeyPrefix = pubKey.testBit(0) ? "03" : "02";
84          byte[] pubKeyBytes = Arrays.copyOf(pubKey.toByteArray(), 32);
85          return new Secp256k1PublicKey(Hex.decode(pubKeyPrefix + Hex.toHexString(pubKeyBytes)));
86      }
87  }
