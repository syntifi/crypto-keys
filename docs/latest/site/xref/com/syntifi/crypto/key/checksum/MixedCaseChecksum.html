3   import org.bouncycastle.jcajce.provider.digest.Keccak;
4   import org.bouncycastle.util.encoders.Hex;
5   
6   /**
7    * Implementation of the EIP-55: Mixed-case checksum address encoding
8    * Documentation available at: https://eips.ethereum.org/EIPS/eip-55
9    *
10   * @author Alexandre Carvalho
11   * @author Andre Bertolace
12   * @since 0.5.0
13   */
14  public class MixedCaseChecksum {
15  
16      public static String kekkac256(String value) {
17          Keccak.DigestKeccak kekkac256 = new Keccak.Digest256();
18          byte[] hash = kekkac256.digest(value.getBytes());
19          return Hex.toHexString(hash);
20      }
21  
22      public static String checksumEncode(String address) {
23          Boolean withPrefix = address.startsWith("0x");
24          String value = address.toLowerCase().replace("0x", "");
25          char[] hash = kekkac256(value).toCharArray();
26          char[] chars = value.toCharArray();
27          char[] encoded = new char[chars.length];
28          for (int i = 0; i < chars.length; i++) {
29              encoded[i] = Character.digit(hash[i], 16) > 7
30                      ? Character.toUpperCase(chars[i])
31                      : chars[i];
32          }
33          return withPrefix
34                  ? "0x" + String.valueOf(encoded)
35                  : String.valueOf(encoded);
36      }

38  }
