8    * Mixed checksumEncoding

10   * @author Alexandre Carvalho
11   * @author Andre Bertolace
12   * @since 0.5.0
13   */
14  public class MixedCaseChecksum {
15  
16      /**
17       * Implementation of the EIP-55: Mixed-case checksum address encoding
18       * Documentation available at: https://eips.ethereum.org/EIPS/eip-55
19       *
20       * @param value Hex string
21       * @return
22       */
23      public static String checksumEncodeEIP55(String value) {
24          char[] hash = Hex.encode(Keccak256.digest(value.toLowerCase().getBytes())).toCharArray();
25          char[] chars = value.toCharArray();
26          char[] encoded = new char[chars.length];
27          for (int i = 0; i < chars.length; i++) {
28              encoded[i] = Character.digit(hash[i], 16) > 7
29                      ? Character.toUpperCase(chars[i])
30                      : Character.toLowerCase(chars[i]);
31          }
32          return String.valueOf(encoded);
33      }
34  
35      /**
36       * Implementation of the CEP-57
37       * Documentation available at: https://github.com/casper-network/ceps/blob/master/text/0057-checksummed-addresses.md
38       *
39       * @param value Hex string
40       * @return
41       */
42      public static String checksumEncodeCEP57(String value) {
43          byte[] hash = Blake2b.digest(Hex.decode(value), 32);
44          char[] chars = value.toCharArray();
45          char[] encoded = new char[chars.length];
46          int whichByte = 0;
47          int whichBit = 0;
48          for (int i = 0; i < chars.length; i++) {
49              boolean bitSet = (Byte.toUnsignedInt(hash[whichByte]) & (1<<whichBit)) != 0;
50              if (Character.isAlphabetic(chars[i])) {
51                  encoded[i] = bitSet ? Character.toUpperCase(chars[i]) : Character.toLowerCase(chars[i]);
52                  whichByte = (whichByte + whichBit/7) % hash.length;
53                  whichBit = (whichBit + 1) % 8;
54              } else {
55                  encoded[i] = chars[i];
56              }
57          }
58          return String.valueOf(encoded);
59      }
