21  /**
22   * ed25519 implementation of {@link AbstractPrivateKey}
23   *
24   * @author Alexandre Carvalho
25   * @author Andre Bertolace
26   * @since 0.1.0
27   */
28  @NoArgsConstructor
29  @EqualsAndHashCode(callSuper = true)
30  public class Ed25519PrivateKey extends AbstractPrivateKey {
31  
32      private Ed25519PrivateKeyParameters privateKeyParameters;
33  
34      public Ed25519PrivateKey(byte[] privateKey) {
35          super(privateKey);
36          loadPrivateKey(privateKey);
37      }
38  
39      @Override
40      public void loadPrivateKey(byte[] privateKey) {
41          privateKeyParameters = new Ed25519PrivateKeyParameters(privateKey, 0);
42      }
43  
44      /*
45       * SEQUENCE (3 elem) INTEGER 0 SEQUENCE (1 elem) OBJECT IDENTIFIER 1.3.101.112
46       * curveEd25519 (EdDSA 25519 signature algorithm) OCTET STRING (32 byte)
47       * 38AECE974291F14B5FEF97E1B21F684394120B6E7A8AFB04398BBE787E8BC559 OCTET STRING
48       * (32 byte) 38AECE974291F14B5FEF97E1B21F684394120B6E7A8AFB04398BBE787E8BC559
49       */
50      @Override
51      public void readPrivateKey(String filename) throws IOException {
52          ASN1Primitive key = ASN1Primitive.fromByteArray(PemFileHelper.readPemFile(filename));
53          PrivateKeyInfo keyInfo = PrivateKeyInfo.getInstance(key);
54          String algoId = keyInfo.getPrivateKeyAlgorithm().getAlgorithm().toString();
55          if (algoId.equals(ASN1Identifiers.Ed25519OID.getId())) {
56              privateKeyParameters = new Ed25519PrivateKeyParameters(keyInfo.getPrivateKey().getEncoded(), 4);
57              setKey(privateKeyParameters.getEncoded());
58          }
59      }
60  
61      @Override
62      public void writePrivateKey(String filename) throws IOException {
63          DERSequence derPrefix = new DERSequence(ASN1Identifiers.Ed25519OID);
64          DEROctetString key = new DEROctetString(new DEROctetString(getKey()));
65          ASN1EncodableVector vector = new ASN1EncodableVector();
66          vector.add(new ASN1Integer(0));
67          vector.add(derPrefix);
68          vector.add(key);
69          DERSequence derKey = new DERSequence(vector);
70          PemFileHelper.writePemFile(filename, derKey.getEncoded(), ASN1Identifiers.PRIVATE_KEY_DER_HEADER);
71      }
72  
73      @Override
74      public byte[] sign(byte[] message) {
75          Signer signer = new Ed25519Signer();
76          signer.init(true, privateKeyParameters);
77          signer.update(message, 0, message.length);
78          byte[] signature;
79          try {
80              signature = signer.generateSignature();
81              return signature;
82          } catch (DataLengthException | CryptoException e) {
83              // TODO: throw new SomeException();
84              return null;
85          }
86      }
87  
88      @Override
89      public AbstractPublicKey derivePublicKey() {
90          return new Ed25519PublicKey(privateKeyParameters.generatePublicKey().getEncoded());
91      }
92  
93      public static Ed25519PrivateKey deriveFromSeed(byte[] seed, int[] path) throws IOException {
94          byte[] init = "ed25519 seed".getBytes(StandardCharsets.UTF_8);
95          byte[] key = HierarchicalDeterministicKey.getFromSeed(seed, init, path);
96          return new Ed25519PrivateKey(key);
97      }
98  
99      public static Ed25519PrivateKey deriveRandomKey() {
100         SecureRandom rnd = new SecureRandom();
101         Ed25519KeyPairGenerator keyPairGenerator = new Ed25519KeyPairGenerator();
102         keyPairGenerator.init(new Ed25519KeyGenerationParameters(rnd));
103         AsymmetricCipherKeyPair asymmetricCipherKeyPair = keyPairGenerator.generateKeyPair();
104         Ed25519PrivateKeyParameters privateKeyParameters = (Ed25519PrivateKeyParameters) asymmetricCipherKeyPair.getPrivate();
105         return new Ed25519PrivateKey(privateKeyParameters.getEncoded());
106     }
